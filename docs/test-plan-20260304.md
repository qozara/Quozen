1. 🛡️ Vulnerability & Gap Analysis

The current testing architecture is fragile and overly reliant on "happy path" component mounting. You have built a decentralized state machine without verifying the UI accurately reflects the underlying mathematical state under duress.

* **Critical (E2E Math Consistency):** The current Playwright functional tests merely check if a string like `'$50.00'` is visible. There is zero programmatic verification that the UI's calculation of "You owe X" matches the `@quozen/core` `Ledger` output. A UI bug could display inverted balances while the database is perfectly fine.  
* **Critical (Concurrency & Auto-Sync):** You implemented Optimistic Concurrency Control (OCC) and Auto-Sync, but your E2E suite does not simulate concurrent background writes to trigger 409 Conflicts or test the `PullToRefresh` state transitions. Your UI is unverified under actual multi-user conditions.  
* **Risk (AI Proxy Integration):** The AI Agent features are tested in isolation via unit tests, but there is no E2E test verifying that the `AgentCommandDrawer` correctly parses a simulated `tool_call` from the mock server and updates the UI.  
* **Risk (Stateless API):** `apps/api` has basic CRUD tests, but lacks tests for boundary conditions, specifically malformed Google OAuth tokens and rate-limiting triggers on the Edge.

import { test, expect } from '@playwright/test';  
import { setupAuth, ensureLoggedIn, resetTestState, setupTestEnvironment } from './utils';  
import { calculateBalances } from '@quozen/core';

test.describe('Feature: Ledger Math & Complete CRUD Lifecycle', () \=\> {  
    test.beforeEach(async ({ page }) \=\> {  
        await resetTestState();  
        await setupTestEnvironment(page.context());  
        await setupAuth(page); // Bypasses auth, sets mock-token-123  
    });

    it('should strictly maintain mathematical consistency across complex expense and settlement flows', async ({ page }) \=\> {  
        // 1\. Initialization  
        await page.goto('/');  
        await ensureLoggedIn(page);

        // 2\. Create Group with Offline Members  
        await page.getByRole('button', { name: 'New Group' }).click();  
        await page.getByLabel('Group Name').fill('Math Validation Trip');  
        await page.getByLabel('Members (Optional)').fill('bob, charlie');  
        await page.keyboard.press('Enter');  
        await page.getByRole('button', { name: 'Create Group' }).click();  
          
        // Dismiss Share Dialog  
        await page.keyboard.press('Escape');  
        await expect(page.getByTestId('header').getByText('Math Validation Trip')).toBeVisible();

        // 3\. Add Complex Expense (User pays $90, split evenly among 3\)  
        await page.getByTestId('button-nav-add').click();  
        await page.getByTestId('input-expense-description').fill('Dinner');  
        await page.getByTestId('input-expense-amount').fill('90');  
        await page.getByTestId('select-category').click();  
        await page.getByRole('option', { name: 'Food & Dining' }).click();  
          
        // Verify UI calculates splits correctly (30 each)  
        await expect(page.getByTestId('input-split-amount-test-user-id')).toHaveValue('30.00');  
        await expect(page.getByTestId('button-submit-expense')).click();  
        await expect(page.getByRole('heading', { name: 'Add Expense' })).not.toBeVisible();

        // 4\. Navigate to Dashboard & Verify Core Math  
        await page.getByTestId('button-nav-home').click();  
        await expect(page).toHaveURL(/.\*dashboard/);

        // Expected state: Me (paid 90, consumed 30\) \= \+60. Bob \= \-30. Charlie \= \-30.  
        await expect(page.getByTestId('text-user-balance')).toHaveTextContent('+60.00');  
          
        // Open Balances  
        await page.getByText('Group Balances').click();  
          
        // Bob's balance should be \-30  
        const bobBalance \= page.getByTestId('text-balance-bob');  
        await expect(bobBalance).toHaveTextContent('-30.00');

        // Charlie's balance should be \-30  
        const charlieBalance \= page.getByTestId('text-balance-charlie');  
        await expect(charlieBalance).toHaveTextContent('-30.00');

        // 5\. Execute Settlement (Bob pays User $30)  
        await page.getByTestId('button-settle-with-bob').click();  
        await expect(page.getByTestId('modal-settlement')).toBeVisible();  
          
        // Verify suggestion logic populated correctly  
        await expect(page.getByTestId('input-settlement-amount')).toHaveValue('30.00');  
        await page.getByTestId('button-record-payment').click();  
        await expect(page.getByTestId('modal-settlement')).not.toBeVisible();

        // 6\. Verify Post-Settlement State  
        // User should now be \+30 (Charlie still owes 30\)  
        await expect(page.getByTestId('text-user-balance')).toHaveTextContent('+30.00');  
        await expect(page.getByTestId('text-balance-bob')).toHaveTextContent('0.00');  
        await expect(page.getByTestId('text-balance-charlie')).toHaveTextContent('-30.00');

        // 7\. Verify Activity Hub UI Reflects Transactions  
        await page.getByTestId('button-nav-expenses').click();  
          
        // Expenses Tab  
        await expect(page.getByText('Dinner')).toBeVisible();  
          
        // Transfers Tab  
        await page.getByRole('tab', { name: 'Transfers' }).click();  
        await expect(page.getByText('$30.00')).toBeVisible();  
    });

    it('should fail gracefully and prevent submission when splits do not match total amount', async ({ page }) \=\> {  
        await page.goto('/groups');  
        await ensureLoggedIn(page);  
          
        // Assume group exists from previous setup or mock state  
        await page.getByRole('button', { name: 'New Group' }).click();  
        await page.getByLabel('Group Name').fill('Validation Group');  
        await page.getByRole('button', { name: 'Create Group' }).click();  
        await page.keyboard.press('Escape');

        // Attempt malformed expense  
        await page.getByTestId('button-nav-add').click();  
        await page.getByTestId('input-expense-description').fill('Bad Math');  
        await page.getByTestId('input-expense-amount').fill('100');  
        await page.getByTestId('select-category').click();  
        await page.getByRole('option', { name: 'Other' }).click();

        // Force split to be wrong  
        await page.getByTestId('input-split-amount-test-user-id').fill('10');  
          
        await page.getByTestId('button-submit-expense').click();

        // System MUST reject  
        const toast \= page.getByText("Split amounts don't match expense amount.");  
        await expect(toast).toBeVisible();  
          
        // Drawer MUST remain open  
        await expect(page.getByRole('heading', { name: 'Add Expense' })).toBeVisible();  
    });  
});
